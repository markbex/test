{
	var f = document.frmemail;
	var _emailhtml = window.frames["_email"].document.body.innerHTML;
	var inblock = 102399;
	for(var i=1;i<=10;i++)

		f.elements("emailhtml"+i).value="";

		{
			f.elements("emailhtml"+i).value = _emailhtml.substr( (i-1)*inblock, inblock );

	}
}

CREATE  PROCEDURE dbo.spd_eml_trace
	@emailid int,
	@type varchar(10)='',
	@key1 varchar(10)='',
	@key2 varchar(20)='',
	@senduser varchar(20)='',
	@sendempno char(3)='',
	@subject varchar(200)='',
	@to varchar(300)='',
	@cc varchar(300)='',
	@bcc varchar(300)='',
	@emailbody varchar(6000)='',
	@msghtml varchar(3000)=''
	
AS
	set nocount on 
	If @emailid=0 
	begin
		
		insert into tb_emailtrace(type,key1,key2,senduser,sendempno,emailbody,createdate,msghtml)
		values(@type,@key1,@key2,@senduser,@sendempno,'',getdate(),@msghtml)

		select @@identity as emailid
		
	end
	Else
	begin
		DECLARE @ptrval binary(16)

		SELECT @ptrval = TEXTPTR(emailbody) 
		FROM tb_emailtrace
		WHERE emailid= @emailid
		
		UPDATETEXT tb_emailtrace.emailbody @ptrval NULL  NULL @emailbody

		update tb_emailtrace
		set subject = @subject,[to]=@to,cc=@cc,bcc=@bcc
		where emailid = @emailid

		select @emailid as emailid
	end
GO
SET QUOTED_IDENTIFIER OFF 
GO
SET ANSI_NULLS ON 
GO
